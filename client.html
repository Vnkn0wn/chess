import java.awt.*;  //abstract windowing toolkit
import java.net.*;  //networking
import java.io.*;   //DataIOStreams

public class Client extends Frame
{TextArea t=new TextArea("tekst\n");

  public Client()
   {//de standaard initialisaties
    t.setBounds(10,200,380,150);
    add(t);
   }

  public void runClient()
   {Socket s;
    DataOutputStream o;
    DataInputStream i;
    try{System.out.println("Client is running");
        s=new Socket(InetAddress.getLocalHost(),5500);
 //     s=new Socket(InetAddress.getByName("j.cs.utwente.nl"),5500);
 //     s=new Socket("130.89.220.2",5500);
        t.append("To: "+s.getInetAddress().getHostName()+"\n");
        i=new DataInputStream(s.getInputStream());
        o=new DataOutputStream(s.getOutputStream());
        t.append("Received: "+i.readUTF()+"\n");
        o.writeUTF("Bye and see you again soon");
        s.close();
       }
    catch (IOException ex)
      {ex.printStackTrace();
      }
   t.append("bye\n");
   }

  class MyWindow extends java.awt.event.WindowAdapter
    {//de standaard window-closing
    }

  public static void main(String[] args)
   {System.out.println("eerst de server runnen, daarna de client");
    Client c =new Client();
    c.runClient();
   }
 }
